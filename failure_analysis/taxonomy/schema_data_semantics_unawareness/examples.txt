for c in candidates:
    bid = c['book_id']
    book = book_map.get(bid, {})
    # search for 'English' in several fields
    text_blob = ' '.join([str(book.get('categories','')), str(book.get('description','')), str(book.get('author','')), str(book.get('details','')), str(book.get('features',''))])
    is_english = 'English' in text_blob or 'english' in text_blob
    results.append({
        'book_id': bid,
        'title': book.get('title'),
        'author': book.get('author'),
        'categories': book.get('categories'),
        'description': book.get('description'),
        'details': book.get('details', ''),
        'features': book.get('features', ''),
        'detected_english': is_english
    })

# Filter to those detected as English
english_books = [r for r in results if r['detected_english']]

# If none detected, fallback: assume all candidates are English
if not english_books:
    english_books = results

# Prepare final compact output: book_id, title, author (as string), categories
final = []

    # author may be JSON string or None
    author = r['author']
